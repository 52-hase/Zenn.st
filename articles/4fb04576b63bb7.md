### app/controllers/application_controller.rb

```rb
# ==============================================================================
# 【詳細解説】ApplicationController について
# ==============================================================================
# ApplicationControllerは、アプリ内にあるすべてのコントローラーの「親クラス」です。
# ここに書かれたルールやメソッドは、他のすべてのコントローラー（子クラス）に自動的に
# 引き継がれ、共通して利用できます。
#
# このファイルには、実務において非常に重要となる以下の3つの仕組みが凝縮されています。
# 1. CSRF対策（セキュリティの要）
# 2. Deviseのパラメーター拡張（認証のカスタマイズ）
# 3. ログイン状態のシミュレート（ActionCableや背景処理の救世主）
# ==============================================================================

class ApplicationController < ActionController::Base
  # ----------------------------------------------------------------------------
  # 1. CSRF対策（クロスサイトリクエストフォージェリ：サイト間リクエスト偽造の防止）
  # ----------------------------------------------------------------------------
  # [仕組み]
  #   Railsはユーザーがフォームを表示した際、推測不可能な暗号化トークン
  #   （authenticity_token）を発行して画面に埋め込みます。ユーザーがデータを送信
  #   （POSTやPATCHなど）した際、Railsはこのトークンが正しいかを自動でチェックします。
  # [なぜ必要なのか]
  #   悪意のある別サイトを開いた際に、ブラウザの仕組みを悪用して、あなたのアプリに
  #   向けて勝手に「パスワード変更」や「退会」などのリクエストを送信させられる被害
  #   （なりすまし操作）を防ぐためです。
  # [with: :exception の意味]
  #   もしトークンが一致しない、または含まれていない不正なリクエストを検知した場合、
  #   セッションをクリアするだけでなく「強制的にエラー（例外）を発生させて」処理を
  #   完全にブロックします。
  # ----------------------------------------------------------------------------
  protect_from_forgery with: :exception


  # ----------------------------------------------------------------------------
  # 2. Deviseのパラメーター拡張（認証のカスタマイズ）
  # ----------------------------------------------------------------------------
  # [背景]
  #   Deviseはデフォルト状態では、セキュリティ（不正なデータ注入の防止）の観点から
  #   「email」と「password」以外の入力項目を一切受け付けない鉄壁のガード
  #   （ストロングパラメーター）が敷かれています。
  # [if: :devise_controller?]
  #   アプリ全体の通常の画面で毎回この処理を動かすのは無駄なため、
  #   「Deviseが提供する画面（ログインや新規登録など）が動く時だけ」に実行を限定するフィルターです。
  # ----------------------------------------------------------------------------
  before_action :configure_permitted_parameters, if: :devise_controller?

  def configure_permitted_parameters
    # [devise_parameter_sanitizer]
    #   Devise独自のパラメーター許可用メソッドです。
    #   「.permit(:sign_up, keys: [:name])」と書くことで、
    #   「新規登録（sign_up）の時だけは、デフォルト項目に加えて、特別に『name（名前）』
    #   カラムもデータベースに保存していいよ」と許可を与えています。
    devise_parameter_sanitizer.permit(:sign_up, keys: [ :name ])
  end


  # ----------------------------------------------------------------------------
  # 3. ログイン状態のシミュレート（ActionCableや背景処理の救世主）
  # ----------------------------------------------------------------------------
  # [なぜこの特殊なメソッドが必要なのか？]
  #   通常のRailsのコントローラーは、ブラウザから「画面を開いて！」というリクエスト（通信）を
  #   受けて動作するため、ビューの中で「current_user（現在ログインしている人）」という
  #   便利なメソッドが使えます。
  #
  #   しかし、ActionCable（非同期チャット）やActiveJob（裏側の自動処理）には、ブラウザからの
  #   直接のリクエストがありません。そのため、チャットの裏側で「メッセージのHTML（部分テンプレート）」
  #   を組み立てようとしても、「今誰がログインしているのか」がシステム側で判断できずエラーになります。
  #   それを解決するために、「強制的に特定のユーザーがログインしている偽の環境」をその場で
  #   偽造するのがこのクラスメソッドです。
  # ----------------------------------------------------------------------------
  def self.render_with_signed_in_user(user, *args)
    # [行①]
    # Deviseの裏側で動いている認証エンジン「Warden（ワーデン）」のデータを、
    # Railsの画面生成システム（Renderer）に持ち込めるようにするための翻訳設定です。
    ActionController::Renderer::RACK_KEY_TRANSLATION["warden"] ||= "warden"
    
    # [行②]
    # 中身が空の偽のWarden環境を作成し、そこに引数で渡された「user」をセットします。
    # これにより、プログラム上で「このユーザーがログインしていますよ」という既成事実（身代わり環境）を作ります。
    proxy = Warden::Proxy.new({}, Warden::Manager.new({})).tap { |i| i.set_user(user, scope: :user) }
    
    # [行③]
    # リクエストなしでHTMLを作れるRails標準の機能「renderer」に、先ほど作った
    # 「ログイン状態の身代わり（proxy）」を合体させて、新しいレンダラーを生成します。
    renderer = self.renderer.new("warden" => proxy)
    
    # [行④]
    # これによって、ビュー（テンプレート）側で「current_user」や「自分と他人でチャットの左右の
    # 見た目を変えるロジック」が呼ばれたとしても、偽環境の「user」に基づいて正しいHTMLが
    # エラーなく生成され、チャット参加者全員へ一斉配信（ActionCable等で利用）できるようになります。
    renderer.render(*args)
  end
end

# ==============================================================================
# ➔ まとめ
# ApplicationController は、
#   1. 悪意のある攻撃（CSRF）からアプリを守り、
#   2. ユーザー登録時に「名前（name）」を保存できるように枠を広げ、
#   3. チャットなどのリアルタイム通信の裏側で「ログイン状態の見た目」を正確に作り出す、
# という、アプリケーション全体の安全・カスタム・先進的な機能を一手に引き受ける「司令塔」です。
# ==============================================================================
```

```rb
# ApplicationController（共通ルールを管理する親クラス）を継承しています。
class LiveEventsController < ApplicationController
  
  # ライブイベントの一覧を表示するためのアクションです。
  # ブラウザで /live_events にアクセスした際に実行されます。
  def index
    # データベースから全てのライブイベントのデータを取り出し、
    # インスタンス変数（@live_events）に代入してビューに渡します。
    @live_events = LiveEvent.all
  end

  # 特定のライブイベントの情報を詳細に表示するためのアクションです。
  # ブラウザで /live_events/1 のようにIDを指定してアクセスした際に実行されます。
  def show
    # URLに含まれるID（params[:id]）を元に、
    # データベースから該当する1件のライブイベントを検索して取得します。
    @live_event = LiveEvent.find(params[:id])
  end
end
```

```rb
# ApplicationController（共通設定）を継承した、ルーム管理用コントローラー
class LiveRoomsController < ApplicationController
  
  # ルームの一覧を表示するためのアクション
  def index
    # 全てのルームデータを取得して、一覧画面（index.html.erb）へ渡します
    @live_rooms = LiveRoom.all
  end

  # 特定のルームを表示するためのアクション
  def show
    # URLのID（例: /live_rooms/5 なら 5）を元に、該当するルームを1件取得
    @live_room = LiveRoom.find(params[:id])
    
    # そのルームに紐づくメッセージを取得します
    # .includes(:user) : 「N+1問題」を避けるため、投稿したユーザー情報もまとめて取得（高速化）
    # .order(created_at: :asc) : メッセージを古い順（作成日時が昇順）に並べて表示
    @messages = @live_room.messages.includes(:user).order(created_at: :asc)
  end

  # 新しいルームを保存するためのアクション
  def create
    # 下のprivateにある「live_room_params」を使って安全にデータを受け取ります
    @live_room = LiveRoom.new(live_room_params)
    
    if @live_room.save
      # 保存に成功した場合、メッセージを表示してそのルームの個別ページへ移動
      flash[:notice] = "ルームが作成されました。"
      redirect_to @live_room
    else
      # 保存に失敗（名前が空欄など）した場合、新規作成画面（new）を再表示します
      render :new
    end
  end

  private

  # 【ストロングパラメーター】
  # セキュリティのため、Webから送られてきたデータのうち「name」だけを受け取るよう制限します
  def live_room_params
    params.require(:live_room).permit(:name)
  end
end
```

### app/controllers/musics_controller.rb
```rb
# ApplicationController（共通設定）を継承した、音楽（楽曲）管理用コントローラー
class MusicsController < ApplicationController
  
  # 楽曲一覧を取得するためのアクション
  # アーティスト別、または「特定のアーティストの特定のライブ」別に曲を表示します
  def index
    # URLパラメーターからアーティスト情報を取得します
    @artist = Artist.find(params[:artist_id])
    
    # もしURLに live_event_id が含まれていれば、そのライブイベント情報を取得します
    @live_event = LiveEvent.find(params[:live_event_id]) if params[:live_event_id].present?

    # 【取得条件の分岐】
    if @live_event
      # ライブイベントが指定されている場合：そのアーティストの曲の中から、そのライブに関連するものだけを抽出
      @musics = @artist.musics.where(live_event_id: @live_event.id)
    else
      # ライブイベントの指定がない場合：そのアーティストの曲を全て取得
      @musics = @artist.musics
    end

    # 【データの整形】
    # 取得した曲データを、ViewやAPIで扱いやすいようにハッシュ形式の配列に変換します
    @musics = @musics.map do |music|
      {
        name: music.name,              # 曲名
        artist: music.artist_name,      # アーティスト名
        id: music.spotify_track_id,    # Spotifyの楽曲識別ID
        live_event_id: music.live_event_id # 関連するライブイベントのID
      }
    end
  end

  # 特定の楽曲の詳細を取得するためのアクション
  def show
    # 親となるアーティストを取得
    @artist = Artist.find(params[:artist_id])
    
    # 通常のIDではなく、SpotifyのトラックID（spotify_track_id）を使って楽曲を1件検索します
    @music = Music.find_by(spotify_track_id: params[:id])
  end
end
```

### app/controllers/users_controller.rb
```rb
class UsersController < ApplicationController
  # 指定したIDのユーザー情報を取得して詳細画面を表示するアクション
  def show
    # URLのパスから取得したID（params[:id]）を元に、データベースからユーザーを探します。
    @user = User.find(params[:id])
  end

  # ユーザー情報を更新するアクション
  def update
    # 更新対象のユーザーをデータベースから特定します。
    @user = User.find(params[:id])

    # チェックボックスなどで「アバターを削除（remove_avatar）」が選択されている場合の処理
    if params[:remove_avatar] == "on"
      # アバター画像が添付されている場合、Active Storageのファイルを物理削除します。
      @user.avatar.purge if @user.avatar.attached?
    end

    # フォームから送られてきたパラメーター（user_params）を使って更新を試みます。
    if @user.update(user_params)
      # 更新に成功した場合は、フラッシュメッセージをセットして詳細画面へリダイレクトします。
      flash[:notice] = "プロフィールが更新されました。"
      redirect_to @user
    else
      # バリデーションエラーなどで更新に失敗した場合は、showテンプレートを再描画（レンダリング）します。
      # ※ redirect_to ではないため、入力内容は保持されたままエラーメッセージが表示されます。
      render "show"
    end
  end

  private

  # ストロングパラメーター：外部からのリクエストのうち、許可する項目を制限します（セキュリティ対策）。
  def user_params
    # userキーの中にある name, email, avatar のみの変更を許可します。
    params.require(:user).permit(:name, :email, :avatar)
  end
end
```
